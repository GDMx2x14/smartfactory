 23    A class for attributing segments of LLM responses to their respective contexts using similarity metrics.

 25    Attributes:

 27        verbose (bool): If True, detailed output is printed for debugging purposes.

 30        context_sentences (List[str]): A list of all unique sentences extracted from the provided context.
 31        sentence_info (Dict[str, Dict]): A dictionary mapping each sentence to its source and original context.
 32        context_embeddings (np.ndarray or None): Precomputed embeddings of context sentences for faster similarity computation.
 33        executor (ThreadPoolExecutor or None): Thread executor for background model loading and operations.
 34        model_future (concurrent.futures.Future or None): Future object containing the loading/loaded model.

 37    def __init__(
 38        self,
 39        context: List[Tuple[str, str]] = [],
 40        threshold: float = 50.0,
 41        verbose: bool = False,
 42        tokenize_context: bool = True,
 43        use_embeddings: bool = True
 44    ):
 45        """
 46        Initializes a RagExplainer object with configurable parameters and starts loading the embedding model if embeddings are used.

 48        Args:
 49            context (List[Tuple[str, str]]): A list of tuples containing (source name, context string).

 53            use_embeddings (bool): If True, embeddings are used for matching responses to context.

 57        self.verbose = verbose
 58        self.tokenize_context = tokenize_context
 59        self.use_embeddings = use_embeddings

 61        # Initialize storage for context-related data
 62        self.context_sentences = []  # Stores unique context sentences
 63        self.sentence_info = {}  # Maps sentences to metadata (source name and original context)
 64        self.context_embeddings = None  # Stores sentence embeddings if embeddings are used

 66        # Load the embedding model only if embedding-based matching is enabled
 67        if self.use_embeddings:
 68            model_path = './models/sentence-transformers/paraphrase-multilingual-MiniLM-L12-v2'

 70            self.executor = ThreadPoolExecutor()
 71            self.model_future = self.executor.submit(

 74        else:
 75            self.executor = None
 76            self.model_future = None

180    def _parse_json_context(self, ctx: str) -> List[str]:
181        """
182        Parses a JSON-formatted context string into a list of key-value formatted sentences.
183
184        Args:
185            ctx (str): The input string containing JSON data.
186
187        Returns:
188            List[str]: A list of sentences extracted and formatted from the JSON data. Each sentence represents
189            a key-value pair or an object in a structured format.
190
191        Raises:
192            json.JSONDecodeError: If the input string is not a valid JSON object or array.
193        """
194        try:
195            # Attempt to decode the JSON string into a Python object
196            data = json.loads(ctx)
197        except json.JSONDecodeError:
198            # If the input is not valid JSON, return an empty list
199            return []
200
201        # Handle cases where the JSON data is a list
202        if isinstance(data, list):
203            sentences = []  # Initialize a list to hold the formatted strings
204            for item in data:
205                if isinstance(item, dict):  # Check if the list item is a dictionary
206                    formatted = json.dumps(item, indent=0)  # Format the dictionary
207                    # Only include the formatted string if it is sufficiently long
208                    if len(formatted.strip()) >= 10:
209                        sentences.append(formatted)
210                else:
211                    # If the item is not a dictionary, treat it as a plain string
212                    if len(str(item).strip()) >= 10:
213                        sentences.append(str(item))
214            return sentences  # Return the list of formatted sentences
215
216        # Handle cases where the JSON data is a dictionary
217        elif isinstance(data, dict):
218            formatted = json.dumps(data, indent=0)  # Format the dictionary
219            # Return the formatted dictionary as a list if it's sufficiently long
220            if len(formatted.strip()) >= 10:
221                return [formatted]
222            else:
223                return []
224
225        # If the JSON data is neither a list nor a dictionary, return an empty list
226        else:
227            return []

229    def _process_context(self, context: List[Tuple[str, str]]):
230        """
231        Processes and tokenizes the context for internal storage.
232
233        Args:
234            context (List[Tuple[str, str]]): The input context to process.
235
236        Raises:
237            ValueError: If the processed context is empty.
238        """
239        # List to track newly added context sentences
240        new_context_sentences = []
241
242        for idx, (source_name, ctx) in enumerate(context):
243            # Attempt to parse JSON-formatted context
244            json_sentences = self._parse_json_context(ctx)
245
246            # Decide whether to use JSON-parsed sentences or tokenize plain text
247            ctx_strings = json_sentences if json_sentences else (
248                sent_tokenize(ctx) if self.tokenize_context else [ctx]
249            )
250
251            # Ensure the context is not empty after processing
252            if not ctx_strings:
253                raise ValueError(f"The context at index {idx} is empty after tokenization or JSON parsing.")
254
255            # Add each unique sentence to the internal storage
256            for string in ctx_strings:
257                if len(string.strip()) >= 10 and string not in self.sentence_info:
258                    self.context_sentences.append(string)
259                    self.sentence_info[string] = {'source_name': source_name, 'original_context': ctx}
260                    new_context_sentences.append(string)
261
262        # Generate embeddings for new sentences if embeddings are enabled
263        if self.use_embeddings and new_context_sentences:
264            model = self._get_embedding_model()
265            new_embeddings = model.encode(new_context_sentences, convert_to_tensor=True)
266            # Concatenate new embeddings with existing ones
267            self.context_embeddings = (
268                np.concatenate((self.context_embeddings, new_embeddings.cpu()), axis=0)
269                if self.context_embeddings is not None
270                else new_embeddings.cpu()
271            )

288    def _update_references(self, references, seen_references, source_name, context_match, original_context, similarity_score):
289        """
290        Adds a new reference entry if it doesn't already exist.
291
292        Args:
293            references (List[Dict[str, Any]]): List of references to update.
294            seen_references (set): Set of seen references to ensure uniqueness.
295            source_name (str): Name of the source where the context was found.
296            context_match (str): The matched context sentence.
297            original_context (str): The original context text.
298            similarity_score (float): The similarity score of the match.
299
300        Returns:
301            int: The reference number assigned to the new entry.
302        """
303        # Use a tuple of (source_name, context_match) as a unique key
304        ref_key = (source_name, context_match)
305        if ref_key not in seen_references:
306            # Mark this reference as seen
307            seen_references.add(ref_key)
308            # Add a new reference entry
309            new_ref = {
310                "reference_number": len(references) + 1,
311                "context": context_match,
312                "original_context": original_context,
313                "source_name": source_name,
314            }
315            references.append(new_ref)
316
317        # Find and return the reference number for this context
318        for ref in references:
319            if ref["source_name"] == source_name and ref["context"] == context_match:
320                return ref["reference_number"]

322    def _match_with_fuzzy(self, response_segments: List[str]) -> Tuple[str, str, List[Dict[str, Any]]]:
323        """
324        Matches response segments with context using fuzzy string matching.
325
326        Args:
327            response_segments (List[str]): Tokenized segments of the response text.
328
329        Returns:
330            Tuple[str, str, List[Dict[str, Any]]]: Contains:
331                - Response text with added references.
332                - JSON-formatted explanation of references.
333                - List of attribution results.
334        """
335        attribution = []  # Stores matching results for each response segment
336        textResponse = ''  # Final response with references added
337        references = []  # List of reference metadata
338        seen_references = set()  # Tracks unique references
339
340        for response_segment in response_segments:
341            original_segment = response_segment  # Keep a copy of the original segment
342
343            # Use fuzzy matching to find the most similar context sentence
344            match = process.extractOne(
345                response_segment, self.context_sentences, scorer=fuzz.partial_ratio, score_cutoff=self.threshold
346            )
347
348            context_match, similarity_score, source_name, original_context = None, 0, None, None
349            if match:
350                # Extract match details
351                context_match, similarity_score, _ = match
352                info = self.sentence_info.get(context_match, {})
353                source_name = info.get('source_name')
354                original_context = info.get('original_context')
355
356                # Add the reference to the list and update the segment
357                ref_num = self._update_references(
358                    references, seen_references, source_name, context_match, original_context, similarity_score
359                )
360                response_segment = self._insert_reference(response_segment, ref_num)
361            else:
362                original_context = None
363
364            # Append the updated segment to the final response
365            textResponse += response_segment + ' '
366
367            # Save the attribution details
368            attribution.append({
369                "response_segment": original_segment,
370                "context": context_match,
371                "source_name": source_name,
372                "similarity_score": similarity_score,
373                "original_context": original_context,
374            })
375
376            # Print verbose debug information if enabled
377            if self.verbose:
378                self._print_verbose(attribution[-1], textResponse.strip(), json.dumps(references, indent=2))
379
380        # Convert the references list to a JSON string
381        textExplanation = json.dumps(references, indent=2)
382        return textResponse.strip(), textExplanation, attribution

384    def _generate_attribution(self, response_segments, similarity_matrix):
385        """
386        Generates attribution results using a similarity matrix.
387
388        Args:
389            response_segments (List[str]): Segments of the response text.
390            similarity_matrix (np.ndarray): Matrix of similarity scores.
391
392        Returns:
393            Tuple[str, str, List[Dict[str, Any]]]: Contains:
394                - Response text with added references.
395                - JSON-formatted explanation of references.
396                - List of attribution results.
397        """
398        attribution = []  # Tracks attribution data for each segment
399        textResponse = ''  # Stores the final response with references
400        references = []  # List of references
401        seen_references = set()  # Tracks unique references
402
403        for idx, response_segment in enumerate(response_segments):
404            original_segment = response_segment  # Keep a copy of the original response segment
405
406            # Retrieve similarity scores for this segment
407            similarities = similarity_matrix[idx]
408            max_similarity = np.max(similarities)  # Maximum similarity score
409            best_match_idx = np.argmax(similarities)  # Index of the best match
410            similarity_score = max_similarity * 100  # Convert to percentage
411
412            context_match, source_name, original_context = None, None, None
413
414            if similarity_score >= self.threshold:
415                # Get the best matching context sentence and its metadata
416                context_match = self.context_sentences[best_match_idx]
417                info = self.sentence_info.get(context_match, {})
418                source_name = info.get('source_name')
419                original_context = info.get('original_context')
420
421                # Add a reference for this match
422                ref_num = self._update_references(
423                    references, seen_references, source_name, context_match, original_context, similarity_score
424                )
425
426                # Append the reference number to the response segment
427                response_segment = self._insert_reference(response_segment, ref_num)
428            else:
429                similarity_score = 0  # No significant match found
430
431            # Append the updated segment to the final response
432            textResponse += response_segment + ' '
433
434            # Save the attribution details
435            attribution.append({
436                "response_segment": original_segment,
437                "context": context_match,
438                "source_name": source_name,
439                "similarity_score": similarity_score,
440                "original_context": original_context,
441            })
442
443            # Print verbose debug information if enabled
444            if self.verbose:
445                self._print_verbose(attribution[-1], textResponse.strip(), json.dumps(references, indent=2))
446
447        # Convert references to JSON
448        textExplanation = json.dumps(references, indent=2)
449        return textResponse.strip(), textExplanation, attribution

181    def _parse_json_context(self, ctx: str) -> List[str]:
182        """
183        Parses a JSON-formatted context string into a list of key-value formatted sentences.
184
185        Args:
186            ctx (str): The input string containing JSON data.
187
188        Returns:
189            List[str]: A list of sentences extracted and formatted from the JSON data. Each sentence represents
190            a key-value pair or an object in a structured format.
191
192        Raises:
193            json.JSONDecodeError: If the input string is not a valid JSON object or array.
194        """
195        try:
196            # Attempt to decode the JSON string into a Python object
197            data = json.loads(ctx)
198        except json.JSONDecodeError:
199            # If the input is not valid JSON, return an empty list
200            return []
201
202        # Handle cases where the JSON data is a list
203        if isinstance(data, list):
204            sentences = []  # Initialize a list to hold the formatted strings
205            for item in data:
206                if isinstance(item, dict):  # Check if the list item is a dictionary
207                    formatted = json.dumps(item, indent=0)  # Format the dictionary
208                    # Only include the formatted string if it is sufficiently long
209                    if len(formatted.strip()) >= 10:
210                        sentences.append(formatted)
211                else:
212                    # If the item is not a dictionary, treat it as a plain string
213                    if len(str(item).strip()) >= 10:
214                        sentences.append(str(item))
215            return sentences  # Return the list of formatted sentences
216
217        # Handle cases where the JSON data is a dictionary
218        elif isinstance(data, dict):
219            formatted = json.dumps(data, indent=0)  # Format the dictionary
220            # Return the formatted dictionary as a list if it's sufficiently long
221            if len(formatted.strip()) >= 10:
222                return [formatted]
223            else:
224                return []
225
226        # If the JSON data is neither a list nor a dictionary, return an empty list
227        else:
228            return []

230    def _process_context(self, context: List[Tuple[str, str]]):
231        """
232        Processes and tokenizes the context for internal storage.
233
234        Args:
235            context (List[Tuple[str, str]]): The input context to process.
236
237        Raises:
238            ValueError: If the processed context is empty.
239        """
240        # List to track newly added context sentences
241        new_context_sentences = []
242
243        for idx, (source_name, ctx) in enumerate(context):
244            # Attempt to parse JSON-formatted context
245            json_sentences = self._parse_json_context(ctx)
246
247            # Decide whether to use JSON-parsed sentences or tokenize plain text
248            ctx_strings = json_sentences if json_sentences else (
249                sent_tokenize(ctx) if self.tokenize_context else [ctx]
250            )
251
252            # Ensure the context is not empty after processing
253            if not ctx_strings:
254                raise ValueError(f"The context at index {idx} is empty after tokenization or JSON parsing.")
255
256            # Add each unique sentence to the internal storage
257            for string in ctx_strings:
258                if len(string.strip()) >= 10 and string not in self.sentence_info:
259                    self.context_sentences.append(string)
260                    self.sentence_info[string] = {'source_name': source_name, 'original_context': ctx}
261                    new_context_sentences.append(string)
262
263        # Generate embeddings for new sentences if embeddings are enabled
264        if self.use_embeddings and new_context_sentences:
265            model = self._get_embedding_model()
266            new_embeddings = model.encode(new_context_sentences, convert_to_tensor=True)
267            # Concatenate new embeddings with existing ones
268            self.context_embeddings = (
269                np.concatenate((self.context_embeddings, new_embeddings.cpu()), axis=0)
270                if self.context_embeddings is not None
271                else new_embeddings.cpu()
272            )

289    def _update_references(self, references, seen_references, source_name, context_match, original_context, similarity_score):
290        """
291        Adds a new reference entry if it doesn't already exist.
292
293        Args:
294            references (List[Dict[str, Any]]): List of references to update.
295            seen_references (set): Set of seen references to ensure uniqueness.
296            source_name (str): Name of the source where the context was found.
297            context_match (str): The matched context sentence.
298            original_context (str): The original context text.
299            similarity_score (float): The similarity score of the match.
300
301        Returns:
302            int: The reference number assigned to the new entry.
303        """
304        # Use a tuple of (source_name, context_match) as a unique key
305        ref_key = (source_name, context_match)
306        if ref_key not in seen_references:
307            # Mark this reference as seen
308            seen_references.add(ref_key)
309            # Add a new reference entry
310            new_ref = {
311                "reference_number": len(references) + 1,
312                "context": context_match,
313                "original_context": original_context,
314                "source_name": source_name,
315            }
316            references.append(new_ref)
317
318        # Find and return the reference number for this context
319        for ref in references:
320            if ref["source_name"] == source_name and ref["context"] == context_match:
321                return ref["reference_number"]

323    def _match_with_fuzzy(self, response_segments: List[str]) -> Tuple[str, str, List[Dict[str, Any]]]:
324        """
325        Matches response segments with context using fuzzy string matching.
326
327        Args:
328            response_segments (List[str]): Tokenized segments of the response text.
329
330        Returns:
331            Tuple[str, str, List[Dict[str, Any]]]: Contains:
332                - Response text with added references.
333                - JSON-formatted explanation of references.
334                - List of attribution results.
335        """
336        attribution = []  # Stores matching results for each response segment
337        textResponse = ''  # Final response with references added
338        references = []  # List of reference metadata
339        seen_references = set()  # Tracks unique references
340
341        for response_segment in response_segments:
342            original_segment = response_segment  # Keep a copy of the original segment
343
344            # Use fuzzy matching to find the most similar context sentence
345            match = process.extractOne(
346                response_segment, self.context_sentences, scorer=fuzz.partial_ratio, score_cutoff=self.threshold
347            )
348
349            context_match, similarity_score, source_name, original_context = None, 0, None, None
350            if match:
351                # Extract match details
352                context_match, similarity_score, _ = match
353                info = self.sentence_info.get(context_match, {})
354                source_name = info.get('source_name')
355                original_context = info.get('original_context')
356
357                # Add the reference to the list and update the segment
358                ref_num = self._update_references(
359                    references, seen_references, source_name, context_match, original_context, similarity_score
360                )
361                response_segment = self._insert_reference(response_segment, ref_num)
362            else:
363                original_context = None
364
365            # Append the updated segment to the final response
366            textResponse += response_segment + ' '
367
368            # Save the attribution details
369            attribution.append({
370                "response_segment": original_segment,
371                "context": context_match,
372                "source_name": source_name,
373                "similarity_score": similarity_score,
374                "original_context": original_context,
375            })
376
377            # Print verbose debug information if enabled
378            if self.verbose:
379                self._print_verbose(attribution[-1], textResponse.strip(), json.dumps(references, indent=2))
380
381        # Convert the references list to a JSON string
382        textExplanation = json.dumps(references, indent=2)
383        return textResponse.strip(), textExplanation, attribution

385    def _generate_attribution(self, response_segments, similarity_matrix):
386        """
387        Generates attribution results using a similarity matrix.
388
389        Args:
390            response_segments (List[str]): Segments of the response text.
391            similarity_matrix (np.ndarray): Matrix of similarity scores.
392
393        Returns:
394            Tuple[str, str, List[Dict[str, Any]]]: Contains:
395                - Response text with added references.
396                - JSON-formatted explanation of references.
397                - List of attribution results.
398        """
399        attribution = []  # Tracks attribution data for each segment
400        textResponse = ''  # Stores the final response with references
401        references = []  # List of references
402        seen_references = set()  # Tracks unique references
403
404        for idx, response_segment in enumerate(response_segments):
405            original_segment = response_segment  # Keep a copy of the original response segment
406
407            # Retrieve similarity scores for this segment
408            similarities = similarity_matrix[idx]
409            max_similarity = np.max(similarities)  # Maximum similarity score
410            best_match_idx = np.argmax(similarities)  # Index of the best match
411            similarity_score = max_similarity * 100  # Convert to percentage
412
413            context_match, source_name, original_context = None, None, None
414
415            if similarity_score >= self.threshold:
416                # Get the best matching context sentence and its metadata
417                context_match = self.context_sentences[best_match_idx]
418                info = self.sentence_info.get(context_match, {})
419                source_name = info.get('source_name')
420                original_context = info.get('original_context')
421
422                # Add a reference for this match
423                ref_num = self._update_references(
424                    references, seen_references, source_name, context_match, original_context, similarity_score
425                )
426
427                # Append the reference number to the response segment
428                response_segment = self._insert_reference(response_segment, ref_num)
429            else:
430                similarity_score = 0  # No significant match found
431
432            # Append the updated segment to the final response
433            textResponse += response_segment + ' '
434
435            # Save the attribution details
436            attribution.append({
437                "response_segment": original_segment,
438                "context": context_match,
439                "source_name": source_name,
440                "similarity_score": similarity_score,
441                "original_context": original_context,
442            })
443
444            # Print verbose debug information if enabled
445            if self.verbose:
446                self._print_verbose(attribution[-1], textResponse.strip(), json.dumps(references, indent=2))
447
448        # Convert references to JSON
449        textExplanation = json.dumps(references, indent=2)
450        return textResponse.strip(), textExplanation, attribution
