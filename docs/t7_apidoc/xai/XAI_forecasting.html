 56    def predict(self, input_data: Union[np.ndarray, torch.Tensor]) -> np.ndarray:
 57        """
 58        Predict a single step ahead given an input sequence.
 59
 60        Handles both PyTorch and sklearn/xgboost models.
 61
 62        For PyTorch models:
 63        - input_data is reshaped to (1, seq_length, 1)
 64        - A torch.Tensor is created and passed through the model in eval mode without gradients.
 65
 66        For sklearn/xgboost models:
 67        - input_data is reshaped to (1, seq_length)
 68        - The model's 'predict' method is called directly.
 69
 70        Args:
 71            input_data (Union[np.ndarray, torch.Tensor]): The input sequence of shape (seq_length,).
 72
 73        Globals:
 74            None
 75
 76        Raises:
 77            None
 78
 79        Returns:
 80            np.ndarray: A 1D numpy array (shape (1,)) representing the model's prediction.
 81        """
 82        # Ensure input_data is a numpy array
 83        if isinstance(input_data, torch.Tensor):
 84            input_data = input_data.detach().cpu().numpy()
 85
 86        if isinstance(self.model, nn.Module):
 87            # PyTorch model prediction
 88            input_data_reshaped = input_data.reshape(1, self.seq_length, 1)
 89            input_tensor = torch.from_numpy(input_data_reshaped).float().to(self.device)
 90            self.model.eval()
 91            with torch.no_grad():
 92                prediction = self.model(input_tensor).cpu().numpy().flatten()
 93        else:
 94            # sklearn/xgboost model prediction
 95            input_data_reshaped = input_data.reshape(1, -1)
 96            prediction = self.model.predict(input_data_reshaped).flatten()
 97
 98        return prediction

100    def predict_with_uncertainty(
101        self,
102        input_data: np.ndarray,
103        n_samples: int = 100,
104        confidence: float = 0.95,
105        step: int = 0
106    ) -> Tuple[float, float, float, float]:
107        """
108        Make a prediction with uncertainty estimation via bootstrapping.
109
110        For PyTorch models:
111        - All perturbed samples are passed in a single batch through the model.
112
113        For sklearn/xgboost models:
114        - The entire perturbed batch is predicted at once using model.predict.
115
116        Args:
117            input_data (np.ndarray): The input sequence of shape (seq_length,).
118            n_samples (int, optional): Number of bootstrap samples. Default is 100.
119            confidence (float, optional): Confidence level for the interval (e.g. 0.95 for 95%). Default is 0.95.
120            step (int, optional): The step number in the autoregressive forecasting sequence. Default is 0.
121
122        Globals:
123            None
124
125        Raises:
126            None
127
128        Returns:
129            Tuple[float, float, float, float]: A tuple containing:
130                mean_pred (float): Mean of bootstrap predictions
131                lower_bound (float): Lower bound of the confidence interval
132                upper_bound (float): Upper bound of the confidence interval
133                probability_in_interval (float): Fraction of predictions within the confidence interval
134        """
135        # Scale noise with step to reflect increasing uncertainty
136        bootstrap_noise_std = self.bootstrap_noise_std_base * (1 + step)
137
138        perturbed_inputs = np.repeat(input_data.reshape(1, -1), n_samples, axis=0) 
139        perturbed_inputs += np.random.normal(0, bootstrap_noise_std, size=perturbed_inputs.shape)
140
141        if isinstance(self.model, nn.Module):
142            # PyTorch model: run batch through model
143            inputs_tensor = torch.from_numpy(perturbed_inputs.reshape(n_samples, self.seq_length, 1)).float().to(self.device)
144            self.model.eval()
145            with torch.no_grad():
146                predictions = self.model(inputs_tensor).cpu().numpy().flatten()
147        else:
148            # sklearn/xgboost model: predict directly on batch
149            predictions = self.model.predict(perturbed_inputs)
150
151        predictions = np.array(predictions)
152        mean_pred = predictions.mean()
153        lower_bound = np.percentile(predictions, ((1 - confidence) / 2) * 100)
154        upper_bound = np.percentile(predictions, (confidence + (1 - confidence) / 2) * 100)
155
156        # Compute the fraction of predictions within the interval
157        within_interval = np.sum((predictions >= lower_bound) & (predictions <= upper_bound))
158        probability_in_interval = within_interval / len(predictions)
159
160        return mean_pred, lower_bound, upper_bound, probability_in_interval

162    def explain_prediction(
163        self,
164        input_data: np.ndarray,
165        input_labels: List[str],
166        num_features: int = 10
167    ) -> List[Tuple[str, float]]:
168        """
169        Generate a LIME explanation for the model's prediction on input_data.
170
171        A new LimeTabularExplainer is instantiated dynamically with updated feature names
172        corresponding to the current input sequence.
173
174        Args:
175            input_data (np.ndarray): Input data of shape (seq_length,).
176            input_labels (List[str]): Labels corresponding to each step in input_data.
177            num_features (int, optional): Number of features to include in the explanation. Default is 10.
178
179        Globals:
180            None
181
182        Raises:
183            None
184
185        Returns:
186            List[Tuple[str, float]]: A list of (feature_label, importance) pairs.
187        """
188        input_data_flat = input_data.flatten()
189
190        def predict_fn(data):
191            batch_size = data.shape[0]
192            if isinstance(self.model, nn.Module):
193                # For PyTorch models, reshape to (batch, seq_length, 1)
194                inputs = data.reshape(batch_size, self.seq_length, 1)
195                inputs_tensor = torch.from_numpy(inputs).float().to(self.device)
196                self.model.eval()
197                with torch.no_grad():
198                    outputs = self.model(inputs_tensor).cpu().numpy()
199            else:
200                outputs = self.model.predict(data)
201            return outputs.flatten()
202
203        # Instantiate a new LimeTabularExplainer for the current labels
204        explainer = LimeTabularExplainer(
205            training_data=self.training_data,
206            feature_names=input_labels,  # Use the current labels directly
207            mode='regression',
208            verbose=False
209        )
210
211        exp = explainer.explain_instance(
212            input_data_flat,
213            predict_fn,
214            num_features=num_features
215        )
216
217        explanation = exp.as_list()
218        return explanation

220    def predict_and_explain(
221        self,
222        input_data: Union[np.ndarray, torch.Tensor],
223        n_predictions: int,
224        input_labels: List[str],
225        num_features: int = 10,
226        confidence: float = 0.95,
227        n_samples: int = 100,
228        use_mean_pred: bool = False
229    ) -> dict:
230        """
231        Perform autoregressive prediction and explanation for n_predictions steps.
232
233        Each predicted step updates the input sequence and labels.
234
235        Args:
236            input_data (Union[np.ndarray, torch.Tensor]): Initial input sequence of shape (seq_length,).
237            n_predictions (int): Number of autoregressive predictions to make.
238            input_labels (List[str]): Labels corresponding to the input_data.
239            num_features (int, optional): Number of features for LIME explanation. Default is 10.
240            confidence (float, optional): Confidence level for interval estimation. Default is 0.95.
241            n_samples (int, optional): Number of bootstrap samples for uncertainty estimation. Must be >=100 for meaningful confidence. Default is 100.
242            use_mean_pred (bool, optional): If True, use mean of bootstrap predictions as final prediction; else use raw prediction. Default is False.
243
244        Globals:
245            None
246
247        Raises:
248            None
249
250        Returns:
251            dict: A dictionary containing:
252                'Predicted_value' (List[float]): List of predicted values
253                'Lower_bound' (List[float]): List of lower bounds
254                'Upper_bound' (List[float]): List of upper bounds
255                'Confidence_score' (List[float]): List of probabilities within the interval
256                'Lime_explaination' (List[List[Tuple[str,float]]]): LIME explanations per step
257                'Date_prediction' (List[str]): Predicted date labels for each step
258        """
259        if isinstance(input_data, torch.Tensor):
260            input_data = input_data.detach().cpu().numpy()
261
262        predicted_values = []
263        lower_bounds = []
264        upper_bounds = []
265        confidence_scores = []
266        lime_explanations = []
267        date_predictions = []
268
269        current_input = input_data.copy()
270        current_labels = input_labels.copy()
271
272        for i in range(n_predictions):
273            # Compute uncertainties and probability in interval
274            mean_pred, lower_bound, upper_bound, probability_in_interval = self.predict_with_uncertainty(
275                current_input, n_samples=n_samples, confidence=confidence, step=i
276            )
277            # Compute raw prediction
278            raw_pred = self.predict(current_input)[0]
279
280            # Decide which prediction to use
281            final_pred = mean_pred if use_mean_pred else raw_pred
282
283            explanation = self.explain_prediction(current_input, current_labels, num_features=num_features)
284
285            predicted_values.append(final_pred)
286            lower_bounds.append(lower_bound)
287            upper_bounds.append(upper_bound)
288            confidence_scores.append(probability_in_interval)
289            lime_explanations.append(explanation)
290
291            # Update the input_data and labels for the next step
292            current_input = np.append(current_input[1:], final_pred)
293            last_label_date = datetime.strptime(current_labels[-1], "%Y-%m-%d")
294            new_label_date = last_label_date + timedelta(days=1)
295            new_label = new_label_date.strftime("%Y-%m-%d")
296            current_labels = current_labels[1:] + [new_label]
297            date_predictions.append(new_label)
298
299        out_dict = {
300            'Predicted_value': predicted_values,
301            'Lower_bound': lower_bounds,
302            'Upper_bound': upper_bounds,
303            'Confidence_score': confidence_scores,
304            'Lime_explaination': lime_explanations,
305            'Date_prediction': date_predictions
306        }
307
308        return out_dict

311def main():
312    """
313    Main entry point to demonstrate the ForecastExplainer class.
314
315    This function:
316    - Generates a synthetic sine wave dataset with noise.
317    - Trains an XGBoost model for forecasting.
318    - Instantiates ForecastExplainer (which internally creates a LimeTabularExplainer).
319    - Uses the ForecastExplainer to predict and explain forecast steps.
320    - Plots the results.
321
322    Args:
323        None
324
325    Globals:
326        None
327
328    Raises:
329        None
330
331    Returns:
332        None
333    """
334    # Seed
335    np.random.seed(42)
336    torch.manual_seed(42)
337    random.seed(42)
338
339    # Generate a sine wave
340    total_points = 300
341    seq_length = 50
342    t = np.linspace(0, 10*np.pi, total_points)
343    data = np.sin(t) + np.random.normal(0, 0.05, size=total_points)
344
345    # Prepare training data for XGBoost: predict next value from last seq_length values
346    X_train = []
347    y_train = []
348    for i in range(total_points - seq_length - 1):
349        X_train.append(data[i:i+seq_length])
350        y_train.append(data[i+seq_length])
351
352    X_train = np.array(X_train)
353    y_train = np.array(y_train)
354
355    # Train XGBoost model
356    model = xgb.XGBRegressor(n_estimators=100, max_depth=3, learning_rate=0.05, random_state=42)
357    model.fit(X_train, y_train)
358
359    # Perform predictions beyond the training range
360    n_predictions = 20
361    input_data = data[(total_points - seq_length - n_predictions): (total_points - n_predictions)]
362
363    # Generate labels for the input_data
364    start_date = datetime(2020, 1, 1)
365    input_labels = [(start_date + timedelta(days=i)).strftime("%Y-%m-%d") for i in range(seq_length)]
366
367    start_time = time.time()
368    # Initialize the explainer
369    explainer = ForecastExplainer(model, X_train)
370
371    # Perform autoregressive predictions
372    results = explainer.predict_and_explain(
373        input_data=input_data,
374        n_predictions=n_predictions,
375        input_labels=input_labels,
376        num_features=5,
377        confidence=0.95,
378        n_samples=100,
379        use_mean_pred=False
380    )
381
382    end_time = time.time()
383
384    print("Results:")
385    for key, val in results.items():
386        if key == 'Lime_explaination':
387            print(f"{key}:")
388            for step, expl in enumerate(val):
389                print(f"  Step {step+1}: {expl}")
390        else:
391            print(f"{key}: {val}")
392
393    # Plot the results as three separate lines
394    predicted_values = results['Predicted_value']
395    lower_bounds = results['Lower_bound']
396    upper_bounds = results['Upper_bound']
397
398    time_indices = np.arange(len(input_data), len(input_data) + n_predictions)
399
400    plt.figure(figsize=(10,6))
401    plt.plot(np.arange(len(input_data)), input_data, label='Input Data', marker='o')
402    plt.plot(time_indices, predicted_values, 'r-', label='Predicted value')
403    plt.plot(time_indices, lower_bounds, 'g--', label='Lower bound')
404    plt.plot(time_indices, upper_bounds, 'b--', label='Upper bound')
405
406    plt.title("Forecasting with XGBoost and Empirical Probability in Interval")
407    plt.xlabel("Time Steps")
408    plt.ylabel("Value")
409    plt.grid(True)
410    plt.legend()
411    plt.show()
412
413    print(f"Time taken: {end_time - start_time:.2f} seconds")

def predict_with_uncertainty( self, input_data: numpy.ndarray, n_samples: int = 100, confidence: float = 0.95, step: int = 0) -> Tuple[float, float, float, float]:

104        n_samples: int = 100,
105        confidence: float = 0.95,
106        step: int = 0
107    ) -> Tuple[float, float, float, float]:
108        """
109        Make a prediction with uncertainty estimation via bootstrapping.
110
111        For PyTorch models:
112        - All perturbed samples are passed in a single batch through the model.
113
114        For sklearn/xgboost models:
115        - The entire perturbed batch is predicted at once using model.predict.
116
117        Args:
118            input_data (np.ndarray): The input sequence of shape (seq_length,).
119            n_samples (int, optional): Number of bootstrap samples. Default is 100.
120            confidence (float, optional): Confidence level for the interval (e.g. 0.95 for 95%). Default is 0.95.
121            step (int, optional): The step number in the autoregressive forecasting sequence. Default is 0.
122
123        Globals:
124            None
125
126        Raises:
127            None
128
129        Returns:
130            Tuple[float, float, float, float]: A tuple containing:
131                mean_pred (float): Mean of bootstrap predictions
132                lower_bound (float): Lower bound of the confidence interval
133                upper_bound (float): Upper bound of the confidence interval
134                probability_in_interval (float): Fraction of predictions within the confidence interval
135        """
136        # Scale noise with step to reflect increasing uncertainty
137        bootstrap_noise_std = self.bootstrap_noise_std_base * (1 + step)
138
139        perturbed_inputs = np.repeat(input_data.reshape(1, -1), n_samples, axis=0) 
140        perturbed_inputs += np.random.normal(0, bootstrap_noise_std, size=perturbed_inputs.shape)
141
142        if isinstance(self.model, nn.Module):
143            # PyTorch model: run batch through model
144            inputs_tensor = torch.from_numpy(perturbed_inputs.reshape(n_samples, self.seq_length, 1)).float().to(self.device)
145            self.model.eval()
146            with torch.no_grad():
147                predictions = self.model(inputs_tensor).cpu().numpy().flatten()
148        else:
149            # sklearn/xgboost model: predict directly on batch
150            predictions = self.model.predict(perturbed_inputs)
151
152        predictions = np.array(predictions)
153        mean_pred = predictions.mean()
154        lower_bound = np.percentile(predictions, ((1 - confidence) / 2) * 100)
155        upper_bound = np.percentile(predictions, (confidence + (1 - confidence) / 2) * 100)
156
157        # Compute the fraction of predictions within the interval
158        within_interval = np.sum((predictions >= lower_bound) & (predictions <= upper_bound))
159        probability_in_interval = within_interval / len(predictions)
160
161        return mean_pred, lower_bound, upper_bound, probability_in_interval
